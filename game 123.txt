<div class="card">
    <div id="skor">Skor: 0</div>
    <div id="box"></div>
    <p>Apa nama warna ini?</p>
    <input type="text" id="jawaban" placeholder="Contoh: red">
    <button onclick="cek()">Cek Jawaban di sini, njing</button>
    <p id="hasil" style="font-size: 14px; margin-top: 15px;"></p>
</div>

<script>
    const warna = ["red", "blue", "green", "yellow", "purple", "orange"];
    let target = "", skor = 0;

    function gantiWarna() {
        target = warna[Math.floor(Math.random() * warna.length)];
        document.getElementById("box").style.backgroundColor = target;
        document.getElementById("hasil").innerHTML = "";
    }

    function cek() {
        let input = document.getElementById("jawaban").value.toLowerCase();
        let hasil = document.getElementById("hasil");
        
        if (input === target) {
            skor++;
            hasil.innerHTML = "✅ Benar Anjing! Lanjut...";
            hasil.style.color = "#059669";
            document.getElementById("skor").innerHTML = "Skor: " + skor;
            setTimeout(gantiWarna, 800);
        } else {
            hasil.innerHTML = "❌ Salah Kontol, coba lagi!";
            hasil.style.color = "#e11d48";
        }
        document.getElementById("jawaban").value = "";
    }

    gantiWarna();
</script>
